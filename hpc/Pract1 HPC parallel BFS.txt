#include <iostream>
#include <vector>
#include <queue>
#include <omp.h>  // For OpenMP
using namespace std;

class Tree {
    int V;  // Number of nodes
    vector<vector<int>> adj;  // Adjacency list
public:
    // Constructor
    Tree(int V) {
        this->V = V;
        adj.resize(V);
    }

    // Add edge (undirected, tree has no cycles)
    void addEdge(int u, int v) {
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    // Parallel BFS
    void parallelBFS(int start) {
        vector<bool> visited(V, false);
        queue<int> q;

        visited[start] = true;
        q.push(start);

        cout << "Parallel BFS starting from node " << start << ":\n";

        while (!q.empty()) {
            int size = q.size();  // Number of nodes at current level
            vector<int> currentLevelNodes;

            // Extract all nodes at this level
            for (int i = 0; i < size; i++) {
                int u = q.front();
                q.pop();
                currentLevelNodes.push_back(u);
            }

            // Process all nodes at current level in parallel
            #pragma omp parallel for
            for (int i = 0; i < currentLevelNodes.size(); i++) {
                int u = currentLevelNodes[i];
                cout << u << " ";

                for (int v : adj[u]) {
                    // Atomically check and mark visited
                    bool expected = false;
                    #pragma omp critical
                    {
                        if (!visited[v]) {
                            visited[v] = true;
                            q.push(v);
                        }
                    }
                }
            }
        }
        cout << endl;
    }
};

int main() {
    // Create a tree with 7 nodes
    Tree t(7);

    // Build the tree
    t.addEdge(0, 1);
    t.addEdge(0, 2);
    t.addEdge(1, 3);
    t.addEdge(1, 4);
    t.addEdge(2, 5);
    t.addEdge(2, 6);

    // Run parallel BFS
    #pragma omp parallel
    {
        #pragma omp single
        {
            t.parallelBFS(0);  // Start BFS from root node 0
        }
    }

    return 0;
}
